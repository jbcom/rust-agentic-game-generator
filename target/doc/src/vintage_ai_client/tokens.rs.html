146impl TokenCounter {
147    /// Create a new token counter
148    pub fn new() -> Self {
149        let mut encoders = HashMap::new();
150
151        // Initialize encoders for different model families
152        encoders.insert("cl100k_base".to_string(), Arc::new(cl100k_base().unwrap()));
153        encoders.insert("p50k_base".to_string(), Arc::new(p50k_base().unwrap()));
154        encoders.insert("r50k_base".to_string(), Arc::new(r50k_base().unwrap()));
155
156        Self {
157            encoders,
158            stats: Arc::new(Mutex::new(TokenStats::default())),
159            pricing: ModelPricing::default(),
160        }
161    }
162
163    /// Count tokens for a given text and model
164    pub fn count_tokens(&self, text: &str, model: &str) -> Result<usize> {
165        let encoder = self.get_encoder_for_model(model)?;
166        let tokens = encoder.encode_with_special_tokens(text);
167        Ok(tokens.len())
168    }
169
170    /// Estimate tokens for an image
171    pub fn estimate_image_tokens(&self, width: u32, height: u32) -> usize {
172        // Rough estimation based on image dimensions
173        // OpenAI doesn't provide exact token counts for images
174        let pixels = width * height;
175        let base_tokens = 85; // Base tokens for any image
176        let dimension_tokens = (pixels as f64 / 750.0).ceil() as usize;
177        base_tokens + dimension_tokens
178    }
179
180    /// Record token usage
181    pub async fn record_usage(
182        &self,
183        model: &str,
184        prompt_tokens: usize,
185        completion_tokens: usize,
186    ) -> Result<()> {
187        let mut stats = self.stats.lock().await;
188
189        // Update token counts
190        stats.prompt_tokens += prompt_tokens as u64;
191        stats.completion_tokens += completion_tokens as u64;
192
193        // Update per-model stats
194        *stats.tokens_by_model.entry(model.to_string()).or_insert(0) +=
195            (prompt_tokens + completion_tokens) as u64;
196
197        // Calculate cost
198        if let Some(pricing) = self.pricing.models.get(model) {
199            let prompt_cost = (prompt_tokens as f64 / 1000.0) * pricing.prompt_cost_per_1k;
200            let completion_cost =
201                (completion_tokens as f64 / 1000.0) * pricing.completion_cost_per_1k;
202            let total_cost = prompt_cost + completion_cost;
203
204            stats.total_cost += total_cost;
205            *stats.cost_by_model.entry(model.to_string()).or_insert(0.0) += total_cost;
206        }
207
208        Ok(())
209    }
210
211    /// Record image generation
212    pub async fn record_image_generation(&self, model: &str, count: usize) -> Result<()> {
213        let mut stats = self.stats.lock().await;
214
215        if let Some(pricing) = self.pricing.models.get(model)
216            && let Some(image_cost) = pricing.image_cost
217        {
218            let total_cost = image_cost * count as f64;
219            stats.total_cost += total_cost;
220            *stats.cost_by_model.entry(model.to_string()).or_insert(0.0) += total_cost;
221            stats.image_tokens += self.estimate_image_tokens(1024, 1024) as u64 * count as u64;
222        }
223
224        Ok(())
225    }
226
227    /// Record embedding usage
228    pub async fn record_embedding(&self, model: &str, tokens: usize) -> Result<()> {
229        let mut stats = self.stats.lock().await;
230
231        stats.embedding_tokens += tokens as u64;
232
233        if let Some(pricing) = self.pricing.models.get(model)
234            && let Some(embedding_cost) = pricing.embedding_cost_per_1k
235        {
236            let total_cost = (tokens as f64 / 1000.0) * embedding_cost;
237            stats.total_cost += total_cost;
238            *stats.cost_by_model.entry(model.to_string()).or_insert(0.0) += total_cost;
239        }
240
241        Ok(())
242    }
243
244    /// Get current statistics
245    pub async fn get_stats(&self) -> TokenStats {
246        self.stats.lock().await.clone()
247    }
248
249    /// Reset statistics
250    pub async fn reset_stats(&self) {
251        *self.stats.lock().await = TokenStats::default();
252    }

254    /// Estimate cost for a request before making it
255    pub fn estimate_cost(
256        &self,
257        model: &str,
258        prompt: &str,
259        max_completion_tokens: usize,
260    ) -> Result<f64> {
261        let prompt_tokens = self.count_tokens(prompt, model)?;
262
263        if let Some(pricing) = self.pricing.models.get(model) {
264            let prompt_cost = (prompt_tokens as f64 / 1000.0) * pricing.prompt_cost_per_1k;
265            let completion_cost =
266                (max_completion_tokens as f64 / 1000.0) * pricing.completion_cost_per_1k;
267            Ok(prompt_cost + completion_cost)
268        } else {
269            Ok(0.0)
270        }
271    }
272
273    /// Get the appropriate encoder for a model
274    fn get_encoder_for_model(&self, model: &str) -> Result<&Arc<CoreBPE>> {
275        // Determine which encoder to use based on model name
276        let encoder_name = if model.starts_with("gpt-4") || model.starts_with("gpt-3.5-turbo") {
277            "cl100k_base"
278        } else if model.contains("davinci") || model.contains("curie") {
279            "p50k_base"
280        } else {
281            "cl100k_base" // Default to newest encoder
282        };
283
284        self.encoders
285            .get(encoder_name)
286            .context(format!("No encoder found for model: {}", model))
287    }

290/// Token optimization strategies
291pub struct TokenOptimizer {
292    /// Maximum context window sizes by model
293    context_windows: HashMap<String, usize>,
294}
295
296impl Default for TokenOptimizer {
297    fn default() -> Self {
298        let mut context_windows = HashMap::new();
299
300        context_windows.insert("gpt-4".to_string(), 8192);
301        context_windows.insert("gpt-4-32k".to_string(), 32768);
302        context_windows.insert("gpt-4-turbo".to_string(), 128000);
303        context_windows.insert("gpt-3.5-turbo".to_string(), 16384);
304        context_windows.insert("gpt-3.5-turbo-16k".to_string(), 16384);
305
306        Self { context_windows }
307    }
308}
309
310impl TokenOptimizer {
311    /// Optimize a prompt to fit within token limits
312    pub fn optimize_prompt(
313        &self,
314        prompt: &str,
315        model: &str,
316        max_completion_tokens: usize,
317        counter: &TokenCounter,
318    ) -> Result<String> {
319        let max_context = self.context_windows.get(model).copied().unwrap_or(4096);
320        let current_tokens = counter.count_tokens(prompt, model)?;
321
322        if current_tokens + max_completion_tokens <= max_context {
323            return Ok(prompt.to_string());
324        }
325
326        // Need to truncate - implement smart truncation
327        let available_tokens = max_context - max_completion_tokens - 100; // Safety margin
328        self.truncate_to_token_limit(prompt, available_tokens, model, counter)
329    }
330
331    /// Truncate text to fit within token limit
332    fn truncate_to_token_limit(
333        &self,
334        text: &str,
335        max_tokens: usize,
336        model: &str,
337        counter: &TokenCounter,
338    ) -> Result<String> {
339        // Binary search for the right truncation point
340        let chars: Vec<char> = text.chars().collect();
341        let mut left = 0;
342        let mut right = chars.len();
343        let mut best_fit = String::new();
344
345        while left < right {
346            let mid = (left + right).div_ceil(2);
347            let truncated: String = chars[..mid].iter().collect();
348            let tokens = counter.count_tokens(&truncated, model)?;
349
350            if tokens <= max_tokens {
351                best_fit = truncated;
352                left = mid;
353            } else {
354                right = mid - 1;
355            }
356        }
357
358        Ok(best_fit)
359    }
360
361    /// Split text into chunks that fit within token limits
362    pub fn chunk_text(
363        &self,
364        text: &str,
365        max_tokens_per_chunk: usize,
366        model: &str,
367        counter: &TokenCounter,
368    ) -> Result<Vec<String>> {
369        let mut chunks = Vec::new();
370        let paragraphs: Vec<&str> = text.split("\n\n").collect();
371        let mut current_chunk = String::new();
372        let mut current_tokens = 0;
373
374        for paragraph in paragraphs {
375            let paragraph_tokens = counter.count_tokens(paragraph, model)?;
376
377            if current_tokens + paragraph_tokens > max_tokens_per_chunk && !current_chunk.is_empty()
378            {
379                chunks.push(current_chunk.clone());
380                current_chunk.clear();
381                current_tokens = 0;
382            }
383
384            if paragraph_tokens > max_tokens_per_chunk {
385                // Split large paragraph
386                let sub_chunks =
387                    self.split_paragraph(paragraph, max_tokens_per_chunk, model, counter)?;
388                for sub_chunk in sub_chunks {
389                    chunks.push(sub_chunk);
390                }
391            } else {
392                if !current_chunk.is_empty() {
393                    current_chunk.push_str("\n\n");
394                }
395                current_chunk.push_str(paragraph);
396                current_tokens += paragraph_tokens;
397            }
398        }
399
400        if !current_chunk.is_empty() {
401            chunks.push(current_chunk);
402        }
403
404        Ok(chunks)
405    }
406
407    fn split_paragraph(
408        &self,
409        paragraph: &str,
410        max_tokens: usize,
411        model: &str,
412        counter: &TokenCounter,
413    ) -> Result<Vec<String>> {
414        let sentences: Vec<&str> = paragraph.split(". ").collect();
415        let mut chunks = Vec::new();
416        let mut current_chunk = String::new();
417        let mut current_tokens = 0;
418
419        for (i, sentence) in sentences.iter().enumerate() {
420            let sentence_with_period = if i < sentences.len() - 1 {
421                format!("{}. ", sentence)
422            } else {
423                sentence.to_string()
424            };
425
426            let sentence_tokens = counter.count_tokens(&sentence_with_period, model)?;
427
428            if current_tokens + sentence_tokens > max_tokens && !current_chunk.is_empty() {
429                chunks.push(current_chunk.trim().to_string());
430                current_chunk.clear();
431                current_tokens = 0;
432            }
433
434            current_chunk.push_str(&sentence_with_period);
435            current_tokens += sentence_tokens;
436        }
437
438        if !current_chunk.is_empty() {
439            chunks.push(current_chunk.trim().to_string());
440        }
441
442        Ok(chunks)
443    }

446/// Cost optimization suggestions
447#[derive(Debug, Clone)]
448pub struct CostOptimizationSuggestion {
449    pub description: String,
450    pub potential_savings: f64,
451    pub implementation_difficulty: Difficulty,
452}
453
454#[derive(Debug, Clone, Copy)]
455pub enum Difficulty {
456    Easy,
457    Medium,
458    Hard,
459}
460
461/// Analyze usage and provide cost optimization suggestions
462pub async fn analyze_usage_for_optimizations(
463    stats: &TokenStats,
464) -> Vec<CostOptimizationSuggestion> {
465    let mut suggestions = Vec::new();
466
467    // Check if using expensive models for simple tasks
468    if let Some(gpt4_cost) = stats.cost_by_model.get("gpt-4")
469        && *gpt4_cost > stats.total_cost * 0.5
470    {
471        suggestions.push(CostOptimizationSuggestion {
472            description: "Consider using GPT-3.5-turbo for simpler tasks".to_string(),
473            potential_savings: gpt4_cost * 0.7,
474            implementation_difficulty: Difficulty::Easy,
475        });
476    }
477
478    // Check prompt efficiency
479    let avg_prompt_length = if stats.prompt_tokens > 0 {
480        stats.prompt_tokens as f64 / stats.tokens_by_model.len() as f64
481    } else {
482        0.0
483    };
484
485    if avg_prompt_length > 1000.0 {
486        suggestions.push(CostOptimizationSuggestion {
487            description: "Optimize prompts to be more concise".to_string(),
488            potential_savings: stats.total_cost * 0.2,
489            implementation_difficulty: Difficulty::Medium,
490        });
491    }
492
493    // Suggest caching for repeated requests
494    suggestions.push(CostOptimizationSuggestion {
495        description: "Enable caching for repeated AI requests".to_string(),
496        potential_savings: stats.total_cost * 0.3,
497        implementation_difficulty: Difficulty::Easy,
498    });
499
500    suggestions
501}
